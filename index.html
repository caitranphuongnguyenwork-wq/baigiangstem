Servo myservo;
int trig = 7, echo = 8;
void setup() {
  pinMode(trig, OUTPUT);
  pinMode(echo, INPUT);
  myservo.attach(9);
  myservo.write(0);
}
void loop() {
  long duration, distance;
  digitalWrite(trig, LOW);
  delayMicroseconds(2);
  digitalWrite(trig, HIGH);
  delayMicroseconds(10);
  digitalWrite(trig, LOW);
  duration = pulseIn(echo, HIGH);
  distance = duration * 0.034 / 2;
  if (distance < 20) {
    myservo.write(90);
    delay(3000);
    myservo.write(0);
  }
  delay(200);
}
